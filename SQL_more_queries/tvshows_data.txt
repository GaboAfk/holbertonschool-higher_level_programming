SELECT
    tv_shows.id,
    tv_shows.title,
    tv_show_genres.show_id,
    tv_show_genres.genre_id,
    tv_genres.id,
    tv_genres.name
FROM
    tv_shows
    JOIN tv_show_genres ON tv_shows.id = tv_show_genres.show_id
    JOIN tv_genres ON tv_show_genres.genre_id = tv_genres.id;
